# Unify

git clone https://github.com/jdcarrothers/unify.git

cd unify

npm i

touch .env

echo "CLIENT_ID=.your truelayer client id here" >> .env
echo "CLIENT_SECRET=.your truelayer secret here" >> .env

npm run dev
